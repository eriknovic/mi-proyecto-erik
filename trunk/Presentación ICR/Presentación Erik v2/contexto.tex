\subsection{Integración de la información de los recursos de información}
\begin{frame}
	\frametitle{Integración de la información de los recursos de información}
	\begin{block}{Definición}
	\justifying
	\small La búsqueda y recuperación significativa de información existente en los recursos de información para responder una consulta dada por un usuario.
	\end{block}
	
	\begin{exampleblock}{Etapas}
	\begin{enumerate}[<+-| alert@+>]
	\item \justifying \small Representar el conocimiento e información de los \textit{recursos de información}.
	\item \justifying \small Buscar y recuperar información, mediante la interrogación de la representación de conocimiento (modelo).
	\end{enumerate}
	\end{exampleblock}
