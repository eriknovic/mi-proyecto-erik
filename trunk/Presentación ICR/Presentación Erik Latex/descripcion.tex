\begin{frame}

	\begin{block}{Hipótesis}
	\justifying 
	\textbf{El uso de las \textit{tecnologías semánticas} es adecuado para lograr la \textit{integración semántica} de \textit{recursos de información} en una \textit{memoria corporativa}}.
	\end{block}

\subsection{Objetivos}
\begin{frame}
	\frametitle{Objetivos} 
	\begin{alertblock}{Objetivo Principal}
	\justifying 
	Contribuir a la \textit{integración semántica} de los \textit{recursos de información} en \textit{una memoria corporativa}, mediante el uso de las \textit{tecnologías semánticas}.
	\end{alertblock}
	
	\begin{block}{Objetivos Particulares}
	\begin{enumerate}[<+-| alert@+>]
	\item \justifying \small Un \textbf{\textit{marco de referencia}} para la \textit{integración semántica} de los \textit{recursos de información}.
	\item \justifying \small Un \textbf{\textit{modelo semántico}} que representa el \textit{conocimiento explícito e implícito} de los \textit{recursos de información}.
	\item \justifying \small Un \textbf{\textit{prototipo de interfaz gráfica de usuario}} que permita a los usuarios consultar y visualizar la información de los recursos de información, interrogando un modelo semántico.
	\item \justifying \small La evaluación de la calidad de los \textbf{\textit{resultados recuperados}} y los \textbf{\textit{tiempos de procesamiento}} de la \textit{integración semántica}.
	\end{enumerate}
	\end{block}
\end{frame}

\subsection{Metodología}
\begin{frame}
	\frametitle{Metodología I}
	\begin{block}{Marco de Referencia}
	\begin{enumerate}
	\item \justifying \small Identificar los \textit{casos de uso}.
	\item \justifying \small Evaluar las \textit{herramientas semánticas}.
	\item \justifying \small Conformar los \textit{recurso de información} de la \textit{memoria corporativa}.
	\end{enumerate}
	
	\begin{exampleblock}{Modelo Semántico}
	\begin{enumerate}
	\setcounter{enumi}{3}
	\item \justifying \small Representar el \textit{conocimiento explícito} de los \textit{recursos de información} en un \textit{modelo semántico} (ontología).
	\item \justifying \small Enriquecer el \textit{modelo semántico} con \textit{reglas de inferencia}.
	\end{enumerate}
	\end{exampleblock}
	
	\begin{enumerate}
	\setcounter{enumi}{5}
	\item \justifying \small Escribir las principales \textit{consultas} en la sintaxis correspondiente.
	\item \justifying \small Emplear un razonador para hacer explícito el conocimiento implícito.
	\item \justifying \small Buscar y recuperar información en la memoria corporativa, interrogando el modelo semántico inferido.
	\end{enumerate}
	\end{block}
\end{frame}

\begin{frame}
	\frametitle{Metodología II}	
	\begin{block}{Prototipo de interfaz gráfica de usuario}	
	\begin{enumerate}
	\setcounter{enumi}{8}
	\item \justifying \small Construir el \textit{prototipo de interfaz de usuario} para la (búsqueda y navegación) de los usuarios en un modelo semántico.
	\end{enumerate}
	\end{block}
	
	\begin{block}{Evaluación}
	\begin{enumerate}
	\setcounter{enumi}{9}
	\item \justifying \small Evaluar la calidad de los resultados con y sin inferencia.
	\item \justifying \small Evaluar los \textit{tiempos promedios} de consulta sobre modelos con/sin inferencia.
	\end{enumerate}
	\end{block}
\end{frame}

\subsection{Aportaciones}
\begin{frame}
	\frametitle{Aportaciones}
	\begin{enumerate}[<+-| alert@+>]
	\item \justifying Un \textit{marco de referencia} para lograr la \textit{integración semántica} de \textit{recursos de información}.
    \item \justifying Un modelo semántico que representa el conocimiento de una memoria corporativa.
    \item \justifying  Un prototipo (interfaz gráfica de usuario) para la interacción amigable (búsqueda y consulta de información) de los usuarios con el modelo semántico.
    \item \justifying Los resultados de nuestra evaluación experimental.
    \item \justifying Un par de scripts para la generación automática y controlada de descripciones (conocimiento explícito) de los \textit{recursos de información}.
	\end{enumerate}
\end{frame}
