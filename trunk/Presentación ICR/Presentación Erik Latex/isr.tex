\section{Integración Semántica de una Memoria Corporativa}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{Marco de Referencia}
\begin{frame}
	\frametitle{Marco de Referencia}
	\begin{block}{Etapas}
		\begin{enumerate}
		\item \justifying \textbf{\textit{Representación del conocimiento explicito de los recursos}} consiste en identificar los recursos de información de la memoria corporativa, así como representar las características y relaciones (conocimiento explícito) de estos recursos en un modelo semántico.
		\item \justifying \textbf{\textit{Enriquecimiento del conocimiento en el modelo}} consiste en introducir reglas de inferencia (axiomas) para completar y enriquecer el modelo semántico con conocimiento implícito del dominio de la memoria corporativa.
		\item \justifying \textbf{\textit{Búsqueda y recuperación de la información en el modelo}} consisten en identificar las principales consultas de los usuarios, así como interrogar el modelo semántico para recuperar la información que responda a estas consultas.
		\end{enumerate}
	\end{block}
\end{frame}

\subsubsection{Herencia de Clases}
\begin{frame}[allowframebreaks]
	\frametitle{Herencia de Clases}
	\begin{block}{Subclase (rdfs:subClassOf)}
	\justifying
	Afirma que una \textit{clase A} se subsume por una \textit{clase B}, es decir, la clase A es un caso particular de la \textit{clase B}. En este caso, las instancias de la clase A son instancias de la clase B.

\subsubsection{Características en las Propiedades}
\begin{frame}
	\frametitle{Características en las propiedades}
	\begin{block}{Propiedad simétrica (owl:SymmetricProperty)}
	\justifying
	Afirma que la \textit{propiedad X} es su propia propiedad inversa, es decir, si la \textit{propiedad X} relaciona al \textit{individuo A} con el \textit{individuo B}, entonces, esta propiedad debe relacionar al \textit{individuo B} con el \textit{individuo A}.
	\end{block}

\subsubsection{Ejecutar las consultas mediante un motor de búsqueda. SPARQL}
\begin{frame}[allowframebreaks]
	\frametitle{Funcionamiento de un motor de búsqueda}
	\begin{block}{Cartografía de Competencias}
	\begin{enumerate}
	\item \justifying Interpreta esta consulta SPARQL dad por un usuario.
	\item \justifying Compara los patrones de la cláusula WHERE con todos los triples en el modelo.
	\item \justifying Recupera la información de las variables resultado.
	\item \justifying Regresa la información de las variables al usuario.
	\end{enumerate}
	\end{block}
