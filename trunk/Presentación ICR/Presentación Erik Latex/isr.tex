\section{Integración Semántica de una Memoria Corporativa}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{Marco de Referencia}
\begin{frame}
	\frametitle{Marco de Referencia}
	\begin{block}{Etapas}
		\begin{enumerate}
		\item \justifying \textbf{\textit{Representación del conocimiento explicito de los recursos}} consiste en identificar los recursos de información de la memoria corporativa, así como representar las características y relaciones (conocimiento explícito) de estos recursos en un modelo semántico.
		\item \justifying \textbf{\textit{Enriquecimiento del conocimiento en el modelo}} consiste en introducir reglas de inferencia (axiomas) para completar y enriquecer el modelo semántico con conocimiento implícito del dominio de la memoria corporativa.
		\item \justifying \textbf{\textit{Búsqueda y recuperación de la información en el modelo}} consisten en identificar las principales consultas de los usuarios, así como interrogar el modelo semántico para recuperar la información que responda a estas consultas.
		\end{enumerate}
	\end{block}
\end{frame}

\subsection{Casos de Uso}
\begin{frame}
	\frametitle{Casos de Uso}
	
	\begin{block}{}
		\begin{itemize}
		\item \justifying \textbf{\textit{Cartografía de Competencias}} consiste en la búsqueda y recuperación de información significativa de las personas a partir de las características personales y profesionales de las mismas.
		\item \justifying \textbf{\textit{Búsqueda de Recursos Digitales}} consiste en la búsqueda y recuperación de información significativa de los documentos y archivos multimedia a partir del contenido de los mismos.
		\end{itemize}

	%%%%%%%%%%%%%%%%%%%%%%%
	\begin{block}{Actividades en la representación del conocimiento}
	\begin{enumerate}
	\item \justifying Asignar un identificador único de recursos para cada recurso de información en la memoria corporativa.
	\item \justifying Asignar los identificadores únicos de recursos a las propiedades.
	\item \justifying Reconocer los valores de las propiedades: otro recurso o literal.
	\item \justifying Generar las tripletas RDF asociadas a las descripciones de los recursos de información.
