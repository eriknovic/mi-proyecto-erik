\section{Problema}
\begin{frame}
	\frametitle{Problema} 
	\begin{exampleblock}{Pregunta investigación}
	\justifying 
	\small ¿Las \textit{tecnologías semánticas} son viables para solucionar la \textit{integración semántica} de los \textit{recursos de información} de una \textit{memoria corporativa}?
	\end{exampleblock}
	
	\begin{figure}
	\includegraphics[scale=0.3]{PreguntaInv} 
	\end{figure}
	
	\begin{block}{Hipótesis}
	\justifying 
	\small \textbf{El uso de las \textit{tecnologías semánticas} es adecuado para lograr la \textit{integración semántica} de \textit{recursos de información} en una \textit{memoria corporativa}}.
	\end{block}
\end{frame}

\subsection{Objetivo General}
\begin{frame}
	\frametitle{Objetivo General}
	\begin{alertblock}{}
	\justifying 
	Contribuir a la \textit{integración semántica} de los \textit{recursos de información} existentes en \textit{una memoria corporativa}, mediante el uso de las \textit{tecnologías semánticas}.
	\end{alertblock}
\end{frame}

\subsection{Objetivos}
\begin{frame}
	\frametitle{Objetivos Particulares} 
	\begin{enumerate}
	\item \justifying \small Proponer un \textit{marco de referencia} para la \textit{integración semántica} de los \textit{recursos de información} existentes en una memoria corporativa.
	\item \justifying \small Proponer un \textbf{modelo semántico} que represente el \textit{conocimiento explícito e implícito} existente en los \textit{recursos de información}.
	\item \justifying \small Implementar un \textbf{prototipo} que permita a los usuarios buscar y recuperar \textit{recursos de información} existentes en una memoria corporativa, así como visualizar las caracterizaciones de estos recursos.
	\end{enumerate}
\end{frame}
