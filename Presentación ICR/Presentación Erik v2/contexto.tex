\subsection{Tecnologías Semánticas}
\begin{frame}
	\frametitle{Tecnologías Semánticas}
	\begin{block}{Definición}
	\justifying 
	\small Las tecnologías semánticas (TS) son un conjunto de metodologías, lenguajes, aplicaciones, herramientas y estándares para suministrar u obtener el significado de las palabras, información y las relaciones entre éstos.
	\end{block}
	
	\begin{figure}
	\includegraphics[scale=0.42]{TSWords} 
	\end{figure}
\end{frame}

\begin{frame}
	\frametitle{Integración Semántica de los Recursos de Información en una Memoria Corporativa}
	\begin{block}{Casos de Uso}
		\begin{itemize}%
		\item \justifying \textbf{\textit{Cartografía de Competencias}} es la búsqueda y recuperación de información significativa de personas a partir de las características personales y profesionales de las mismas.
		\item \justifying \textbf{\textit{Recursos Digitales}} es la búsqueda y recuperación de los documentos basados en texto y archivos multimedia a partir del contenido de los mismos.
		\end{itemize}
	\end{block}
