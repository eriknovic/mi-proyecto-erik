\section{Integración Semántica de una Memoria Corporativa}

\begin{frame}
	\frametitle{Integración Semántica de una Memoria Corporativa}
	\begin{block}{Etapas}
		\begin{enumerate}[<+-| alert@+>]
		\item \justifying Representar las características y/o relaciones de los \textit{recursos de información}, para construir un modelo semántico.
		\item \justifying Introducir \textit{reglas de inferencia} en el modelo, para enriquecer con \textit{conocimiento implícito} de los \textit{recursos de información} y del dominio de la memoria.
		\item \justifying Buscar y recuperar información en el modelo semántico para responder un conjunto consultas.
		\end{enumerate}
	\end{block}

\subsubsection{Representar el conocimiento e información mediante el estándar RDF}
\begin{frame}
	\frametitle{Representar el conocimiento e información mediante el estándar RDF}
	
	\begin{block}{Definición}
	\justifying 
	Marco genérico para describir el conocimiento e información explícita de los recursos mediante sus características y relaciones. \begin{scriptsize}\cite{SurvSemWeb2012}\end{scriptsize}
	\end{block}
	
	\begin{block}{Actividades en la representación del conocimiento}
	\begin{enumerate}[<+-| alert@+>]
	\item \justifying Asignar un \textit{identificador único de recursos} (URI) a cada \textit{recurso de información} en la \textit{memoria corporativa}.
	\item \justifying Asignar un URI a cada característica y/o relación (propiedad) de de los \textit{recursos de información}.
	\item \justifying Generar las tripletas RDF asociadas a las descripciones de los recursos de información.
	\end{enumerate}
	\end{block}

\subsection{Enriquecer el conocimiento en el modelo semántico}
\begin{frame}
	\frametitle{Ontología}
	%%%%%%%%%%%%%%%%%%%%%%%%%%%%
	\begin{alertblock}{Definición}
	\justifying 
	Una definición formal, explícita y compartida de los conceptos, así como las relaciones de un determinado dominio. \begin{scriptsize}\cite{Gruber}\end{scriptsize}
	\end{alertblock}
	
	\begin{block}{Componentes}
	\begin{itemize}
	\item \justifying \textbf{\textit{Componente Asertivo (ABox)}} está constituido por descripciones que afirman que los individuos son instancias de una clase o propiedad.
	\item \justifying \textbf{\textit{Componente Terminológico (TBox)}} describe las clases y propiedades relevantes, así como las reglas de inferencia que permiten aprovechar la manera en que las instancias se relacionan entre sí.
	\end{itemize}
	\end{block}
	%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\end{frame}

\begin{frame}
	\frametitle{Axiomatización}	
	\begin{block}{Reglas de inferencia o Axiomas}
	\justifying 
	Expresiones para enriquecer un grafo RDF con conocimiento implícito.\\
	\end{block}
	
	\begin{block}{Lenguajes}
	Especificaciones para describir clases, propiedades e individuos.
	\begin{itemize}
	\item \justifying \textit{RDF Schema \textbf{RDF(S)}}
	\item \justifying \textit{Web Ontology Language \textbf{OWL}} 
	\end{itemize}
	\end{block}
	
	\begin{figure}
	\includegraphics[scale=0.4]{PrefijosRDFSOWL}
	\end{figure}
	
	\begin{block}
	\justifying
	\textit{Lo que es obvio para un humano, no lo es para una maquina}.
	\end{block}
\end{frame}

\subsubsection{Herencia de Clases}
\begin{frame}[allowframebreaks]
	\frametitle{Herencia de Clases}
	\begin{block}{Subclase (rdfs:subClassOf)}
	\justifying
	Afirma que una \textit{clase A} se subsume por una \textit{clase B}, es decir, la clase A es un caso particular de la \textit{clase B}. En este caso, las instancias de la clase A son instancias de la clase B.
	\end{block}

\subsubsection{Dominio y Rango en las Propiedades}
\begin{frame}[allowframebreaks]
	\frametitle{Dominio y Rango en las Propiedades}
	\begin{block}{Dominio (rdfs:domain)}
	\justifying
	Especifica qué clase se aplica a una propiedad.
	\end{block}
	
	\begin{block}{Rango (rdfs:range)}
	\justifying
	Especifica los valores (clase o tipo de literal) que puede asumir una propiedad.

\subsubsection{Características en las Propiedades}
\begin{frame}
	\frametitle{Características en las propiedades}
	\begin{block}{Propiedad simétrica (owl:SymmetricProperty)}
	\justifying
	Afirma que la \textit{propiedad X} es su propia propiedad inversa, es decir, si la \textit{propiedad X} relaciona al \textit{individuo A} con el \textit{individuo B}, entonces, esta propiedad debe relacionar al \textit{individuo B} con el \textit{individuo A}.
	\end{block}
