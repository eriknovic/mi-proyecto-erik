\subsection{Integración Semántica de Recursos de Información}
\subsubsection{Integración a partir del uso de tecnologías semánticas}
\begin{frame}
	\frametitle{Integración de la Información}
	\begin{block}{\cite{Archetype}}
	\justifying
	Una \textit{arquitectura dual} que representa la información clínica de cualquier persona, para que los profesionales de la salud accedan al historial clínico de las mismas.
	\end{block}
	
	\begin{block}{\cite{JunZhai}}
	\justifying
	Una arquitectura basada en ontologías y el \textit{lenguaje de marcado extensible}, para la \textit{integración semántica} en \textit{sistemas de información de energía eléctrica} (sistemas heterogéneos).
	\end{block}
	
	\begin{block}{\cite{Accommodation}}
	\justifying
	Un \textit{marco de integración semántica} para la integración de la información en el dominio del alojamiento en-línea (información en constante cambio), con el fin de reunir y compartir esta información.
	\end{block}
\end{frame}

\subsubsection{Búsqueda y Recuperación de la Información}
\begin{frame}%[allowframebreaks]
	\frametitle{Búsqueda y Recuperación de la Información}
	\begin{block}{\begin{scriptsize}\cite{Tuan-Dung2011}\end{scriptsize}}
	\justifying
	\small Un sistema para la recomendación y el acceso a la información turística en una ontología que define conceptos y propiedades del dominio del turismo.
	\end{block}
	
	\begin{block}{\begin{scriptsize}\cite{Inay2011}\end{scriptsize}}
	\justifying
	\small Una propuesta para representar y recuperar información en una ontología sobre el contenido de los \textit{manuales de mantenimiento electrónicos}, así como un sistema para visualizar la información en una ontología.
	\end{block}
	
	\begin{block}{\begin{scriptsize}\cite{Suganyakala2013}\end{scriptsize}}
	\justifying
	\small Un \textit{marco de trabajo} para recuperar información en una ontología (Película) y una \textit{interfaz de consulta} para escribir consultas en lenguaje natural.
	\end{block}
	
	\begin{block}{\begin{scriptsize}\cite{Salam2013}\end{scriptsize}}
	\justifying
	\small Un método basado en ontologías para recuperar la \textit{información contenida en un documento} mediante el uso de conceptos de una ontología.
	\end{block}
\end{frame}


\subsubsection{Gestión de una Memoria Corporativa}
\begin{frame}
	\frametitle{Gestión de una Memoria Corporativa}
	\begin{block}{\begin{scriptsize}\cite{WangXin}\end{scriptsize}}
	\justifying
	Un enfoque basado en las \textbf{ontologías}, para describir los \textit{objetos de justificación del diseño}, así como acceder de manera uniforme a los \textit{recursos de información}.
	\end{block}
	
	\begin{block}{\begin{scriptsize}\cite{Behja}\end{scriptsize}}
	\justifying
	Un entorno de colaboración para la gestión de memorias corporativas, cuya función es construir una \textit{memoria corporativa} a partir de \textit{lluvia de ideas} y un \textit{mecanismo de toma de decisiones consensuadas}.
	\end{block}
\end{frame}

\subsection{Herramientas para la Integración Semántica de Recursos de Información}
\begin{frame}
	\frametitle{Herramientas para la Integración Semántica de Recursos de Información}
	\begin{block}{Descriptor Semántico de Recursos}
	\justifying
	\small Herramienta para crear y almacenar tripletas RDF, en varias sintaxis de serialización, a partir de la información explícita de los recursos de información. \textbf{\textit{OntoMat Annotizer}}, \textbf{\textit{MnM}}, \textbf{\textit{GATE}} y \textbf{\textit{Aktive Media}}.
	\end{block}
	
	\begin{block}{Editor de Ontologías}
	\justifying
	\small Herramienta que proporciona una serie de interfaces amigables para la construcción y mantenimiento de ontologías. \textbf{\textit{Protégé}}, \textbf{\textit{pOWL}}, \textbf{\textit{TopBraid Composer}} y \textbf{\textit{SWOOP}}.
	\end{block}
	
	\begin{block}{Triplestore}
	\justifying
	\small Programa para el almacenamiento e indexación de tripletas RDF, con el fin de permitir la consulta eficiente de información sobre estas tripletas. \textbf{\textit{Apache Jena}}, \textbf{\textit{Stardog}}, \textbf{\textit{4store}} y \textbf{\textit{Sesame}}.
	\end{block}
\end{frame}
