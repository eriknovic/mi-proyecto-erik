\section{Estado del Arte}
\begin{frame}%[allowframebreaks]
	\frametitle{Estado del Arte}
	\begin{block}{Ejes claves}
	\begin{enumerate}
	\item \justifying Integración de la información a partir del uso de tecnologías semánticas.
	\item \justifying Búsqueda, recuperación y publicación de la información desde una ontología.
	\item \justifying Gestión de una memoria corporativa.
	\end{enumerate}
	\end{block}
\end{frame}

\subsection{Integración de la Información}
\begin{frame}
	\frametitle{Integración de la Información}
	\begin{block}{\begin{scriptsize}\cite{Archetype}\end{scriptsize}}
	\justifying
	Una \textit{arquitectura dual} que representa la información clínica de cualquier persona, para que los profesionales de la salud accedan al historial clínico de las mismas.
	\end{block}
	
	\begin{block}{\begin{scriptsize}\cite{JunZhai}\end{scriptsize}}
	\justifying
	Una arquitectura basada en ontologías y el \textit{lenguaje de marcado extensible}, para la \textit{integración semántica} en \textit{sistemas de información de energía eléctrica} (sistemas heterogéneos).
	\end{block}
	
	\begin{block}{\begin{scriptsize}\cite{Accommodation}\end{scriptsize}}
	\justifying
	Un \textit{marco de integración semántica} para la integración de la información en el dominio del alojamiento en-línea (información en constante cambio), con el fin de reunir y compartir esta información.
	\end{block}
\end{frame}

\subsection{Búsqueda y Recuperación de la Información}
\begin{frame}[allowframebreaks]
	\frametitle{Búsqueda y Recuperación de la Información}
	\begin{block}{\begin{scriptsize}\cite{Tuan-Dung2011}\end{scriptsize}}
	\justifying
	Un sistema para la recomendación y el acceso a la información turística semántica, mediante el uso de una ontología para definir conceptos y propiedades sobre el turismo.
	\end{block}
	
	\begin{block}{\begin{scriptsize}\cite{Inay2011}\end{scriptsize}}
	\justifying
	Una propuesta para representar y recuperar información sobre los contenidos de los \textit{manuales de mantenimiento electrónicos}, basado en una ontología, así como un \textit{sistema de visualización} de la información de una ontología.
	\end{block}
	
	\begin{block}{\begin{scriptsize}\cite{Suganyakala2013}\end{scriptsize}}
	\justifying
	Un \textit{marco de trabajo} para la recuperación de información basado en ontología (Película), así como el diseño de una \textit{interfaz de consulta} para que un usuario pueda escribir una consulta en lenguaje natural.
	\end{block}
	
	\begin{block}{\begin{scriptsize}\cite{Salam2013}\end{scriptsize}}
	\justifying
	Un método basado en ontologías para recuperar la \textit{información contenida en un documento} a partir de los conceptos que están en una ontología.
	\end{block}
\end{frame}


\subsection{Gestión de una Memoria Corporativa}
\begin{frame}
	\frametitle{Gestión de una Memoria Corporativa}
	\begin{block}{\begin{scriptsize}\cite{WangXin}\end{scriptsize}}
	\justifying
	Un enfoque basado en las \textbf{ontologías}, para describir los \textit{objetos de justificación del diseño}, así como acceder de manera uniforme a los \textit{recursos de información}.
	\end{block}
	
	\begin{block}{\begin{scriptsize}\cite{Behja}\end{scriptsize}}
	\justifying
	Un entorno de colaboración para la gestión de memorias corporativas, cuya función es construir una \textit{memoria corporativa} a partir de \textit{lluvia de ideas} y un \textit{mecanismo de toma de decisiones consensuadas}.
	\end{block}
\end{frame}
