\begin{frame}
	\frametitle{Conclusiones}
	\begin{exampleblock}{Hipótesis}
	\justifying 
	\textbf{El uso de las \textit{tecnologías semánticas} es adecuado para lograr la \textit{integración semántica} de \textit{recursos de información} en una \textit{memoria corporativa}}.
	\end{exampleblock}
	
	\begin{block}{Ventajas de las Tecnologías Semánticas}
	\begin{itemize}
	\item \justifying Modelos en un formato estándar.
	\item \justifying Modelos flexibles, extensibles y reutilizables.
	\item \justifying Uso de Lenguajes estándar (World Wide Web Consortium).
	\item \justifying Modelos con conocimiento explícito e implícito.
	\item \justifying Inferencia para materializar tripletas RDF.
	\item \justifying Aplicaciones genéricas.
	\end{itemize}
	\end{block}
\end{frame}
